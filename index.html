function doPost(e) { var dados = JSON.parse(e.postData.contents); var plan = SpreadsheetApp.getActiveSpreadsheet(); if (dados.acao === "login") { var abaConfig = plan.getSheetByName("config"); if (!abaConfig) return ContentService.createTextOutput(JSON.stringify({"status": "erro", "mensagem": "Aba 'config' não encontrada!"})).setMimeType(ContentService.MimeType.JSON); var senhaCorreta = abaConfig.getRange("B1").getValue().toString().trim(); if (dados.senha === senhaCorreta) { return ContentService.createTextOutput(JSON.stringify({"status": "sucesso"})).setMimeType(ContentService.MimeType.JSON); } else { return ContentService.createTextOutput(JSON.stringify({"status": "erro"})).setMimeType(ContentService.MimeType.JSON); } } var abaLista = plan.getSheetByName("Lista"); var abaBase = plan.getSheetByName("BaseProdutos"); if (dados.acao === "carregarBase") { var baseDados = abaBase.getDataRange().getValues(); if(baseDados.length > 0) baseDados.shift(); return ContentService.createTextOutput(JSON.stringify(baseDados)).setMimeType(ContentService.MimeType.JSON); } if (dados.acao === "adicionar") { var id = new Date().getTime().toString(); var dataComoTexto = "'" + dados.data; abaLista.appendRow([dataComoTexto, dados.categoria, dados.produto, dados.quantidade, "", "", id]); return ContentService.createTextOutput(JSON.stringify({"status": "sucesso"})).setMimeType(ContentService.MimeType.JSON); } if (dados.acao === "cadastrarProdutoBase") { abaBase.appendRow([dados.categoria, dados.produto, "", ""]); return ContentService.createTextOutput(JSON.stringify({"status": "sucesso"})).setMimeType(ContentService.MimeType.JSON); } if (dados.acao === "listar") { var dadosLista = abaLista.getDataRange().getValues(); var resultado = []; for (var i = 1; i < dadosLista.length; i++) { var dataLinha = dadosLista[i][0]; var dataFormatada = String(dataLinha).replace("'", "").trim().substring(0, 10); if (dataLinha instanceof Date) dataFormatada = Utilities.formatDate(dataLinha, plan.getSpreadsheetTimeZone(), "yyyy-MM-dd"); if (dataFormatada === dados.data) { resultado.push({ categoria: dadosLista[i][1], produto: dadosLista[i][2], quantidade: dadosLista[i][3], preco: dadosLista[i][4], mercado: dadosLista[i][5], id: dadosLista[i][6] }); } } return ContentService.createTextOutput(JSON.stringify(resultado)).setMimeType(ContentService.MimeType.JSON); } if (dados.acao === "atualizar") { var dadosLista = abaLista.getDataRange().getValues(); for (var i = 1; i < dadosLista.length; i++) { if (dadosLista[i][6] == dados.id) { abaLista.getRange(i + 1, 5).setValue(dados.preco); abaLista.getRange(i + 1, 6).setValue(dados.mercado); var produtosCadastrados = abaBase.getDataRange().getValues(); for (var j = 1; j < produtosCadastrados.length; j++) { if (produtosCadastrados[j][1].toString().toLowerCase() == dadosLista[i][2].toString().toLowerCase()) { abaBase.getRange(j + 1, 3).setValue(dados.preco); abaBase.getRange(j + 1, 4).setValue(dados.mercado); break; } } return ContentService.createTextOutput(JSON.stringify({"status": "sucesso"})).setMimeType(ContentService.MimeType.JSON); } } } if (dados.acao === "atualizarQtd") { var dadosLista = abaLista.getDataRange().getValues(); for (var i = 1; i < dadosLista.length; i++) { if (dadosLista[i][6] == dados.id) { abaLista.getRange(i + 1, 4).setValue(dados.quantidade); return ContentService.createTextOutput(JSON.stringify({"status": "sucesso"})).setMimeType(ContentService.MimeType.JSON); } } } if (dados.acao === "excluir") { var dadosLista = abaLista.getDataRange().getValues(); for (var i = 1; i < dadosLista.length; i++) { if (dadosLista[i][6] == dados.id) { abaLista.deleteRow(i + 1); return ContentService.createTextOutput(JSON.stringify({"status": "sucesso"})).setMimeType(ContentService.MimeType.JSON); } } } }
